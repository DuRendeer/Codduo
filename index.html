import React, { useState, useEffect } from 'react';
import { View, Text, FlatList, TouchableOpacity } from 'react-native';

const TaskManager = () => {
  const [tasks, setTasks] = useState([]);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    fetchTasks();
  }, []);

  const fetchTasks = async () => {
    try {
      const response = await api.get('/tasks');
      setTasks(response.data);
      setLoading(false);
    } catch (error) {
      console.error('Erro ao buscar tasks:', error);
    }
  };

  const toggleTask = (id) => {
    setTasks(tasks.map(task => 
      task.id === id 
        ? { ...task, completed: !task.completed }
        : task
    ));
  };

  return (
    <View style={styles.container}>
      <Text style={styles.title}>Minhas Tarefas</Text>
      <FlatList
        data={tasks}
        keyExtractor={(item) => item.id.toString()}
        renderItem={({ item }) => (
          <TouchableOpacity onPress={() => toggleTask(item.id)}>
            <Text style={styles.taskText}>{item.title}</Text>
          </TouchableOpacity>
        )}
      />
    </View>
  );
};

import pandas as pd
import smtplib
from datetime import datetime, timedelta

class ReportAutomation:
    def __init__(self, database_url, email_config):
        self.db_url = database_url
        self.email_config = email_config
    
    def generate_daily_report(self):
        # Conecta ao banco de dados
        df = pd.read_sql_query("""
            SELECT 
                DATE(created_at) as data,
                COUNT(*) as total_vendas,
                SUM(valor) as receita_total,
                AVG(valor) as ticket_medio
            FROM vendas 
            WHERE created_at >= %s
            GROUP BY DATE(created_at)
        """, self.db_url, params=[datetime.now() - timedelta(days=1)])
        
        # Gera gráficos e análises
        insights = self.analyze_data(df)
        
        # Envia relatório por email
        self.send_email_report(insights)
        
        return "Relatório enviado com sucesso!"
    
    def analyze_data(self, df):
        return {
            'vendas_hoje': df['total_vendas'].sum(),
            'crescimento': self.calculate_growth(df),
            'top_produtos': self.get_top_products()
        }

// Flutter Delivery App
import 'package:flutter/material.dart';
import 'package:geolocator/geolocator.dart';

class DeliveryTracker extends StatefulWidget {
  @override
  _DeliveryTrackerState createState() => _DeliveryTrackerState();
}

class _DeliveryTrackerState extends State<DeliveryTracker> {
  Position? currentPosition;
  
  @override
  void initState() {
    super.initState();
    getCurrentLocation();
  }

  Future<void> getCurrentLocation() async {
    try {
      Position position = await Geolocator.getCurrentPosition(
        desiredAccuracy: LocationAccuracy.high
      );
      
      setState(() {
        currentPosition = position;
      });
    } catch (e) {
      print('Erro ao obter localização: $e');
    }
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: Text('Rastreamento de Entrega'),
        backgroundColor: Colors.deepPurple,
      ),
      body: currentPosition == null
        ? Center(child: CircularProgressIndicator())
        : GoogleMap(
            initialCameraPosition: CameraPosition(
              target: LatLng(
                currentPosition!.latitude, 
                currentPosition!.longitude
              ),
              zoom: 15.0,
            ),
          ),
    );
  }
}

import React, { useState, useEffect } from 'react';

const AnalyticsDashboard = () => {

    totalUsers: 0,
    conversionRate: 0,
    revenue: 0,
    growth: 0
  });
